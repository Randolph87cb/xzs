int knapsack1D(int W, vector<int>& wt, vector<int>& val, int n) {

    for (int i = 0; i < n; ++i) {
        for (int w = W; w >= wt[i]; --w) {
            dp[w] = max(dp[w], dp[w - wt[i]] + val[i]);
        }

}
