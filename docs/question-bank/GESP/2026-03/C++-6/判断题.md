struct Node {

    Node(int v) : val(v), left(nullptr), right(nullptr) {}

void travel(Node* root) {
    if (!root) return;
    stack<Node*> s;
    s.push(root);

    while (!s.empty()) {
        Node* cur = s.top(); s.pop();
        cout << cur->val << " ";

        if (cur->right) s.push(cur->right);
        if (cur->left)  s.push(cur->left);
    }
}
